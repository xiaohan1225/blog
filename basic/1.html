const Placement = 0b001;
const Update = 0b010;

let flags = 0b000;

// 增加操作
flags |= Placement;
flags |= Update;

// 减少操作
flags &= ~Placement;

// 判断是否包含某个操作
if (flags & Placement === Placement) {
  // do something
}
if (flags & Update === Update) {
  // do something
}

// 判断是不包含某个操作
if (flags & Placement === 0) {
  // do something
}
if (flags & Update === 0) {
  // do something
}
